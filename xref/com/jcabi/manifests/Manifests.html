56  /**
57   * Static reader of {@code META-INF/MANIFEST.MF} files.
58   *
59   * The class provides convenient methods to read
60   * all {@code MANIFEST.MF} files available in classpath
61   * and all attributes from them. This mechanism is very useful for transferring
62   * information from continuous integration environment to the production
63   * environment. For example, you want your site to show project version and
64   * the date of {@code WAR} file packaging. First, you configure
65   * {@code maven-war-plugin} to add this information to {@code MANIFEST.MF}:
66   *
67   * <pre> &lt;plugin>
68   *  &lt;artifactId>maven-war-plugin&lt;/artifactId>
69   *  &lt;configuration>
70   *   &lt;archive>
71   *    &lt;manifestEntries>
72   *     &lt;Foo-Version>${project.version}&lt;/Foo-Version>
73   *     &lt;Foo-Date>${maven.build.timestamp}&lt;/Foo-Date>
74   *    &lt;/manifestEntries>
75   *   &lt;/archive>
76   *  &lt;/configuration>
77   * &lt;/plugin></pre>
78   *
79   * <p>{@code maven-war-plugin} will add these attributes to your
80   * {@code MANIFEST.MF} file and the
81   * project will be deployed to the production environment. Then, you can read
82   * these attributes where it's necessary (in one of your JAXB annotated objects,
83   * for example) and show to users:
84   *
85   * <pre> import com.jcabi.manifests.Manifest;
86   * import java.text.SimpleDateFormat;
87   * import java.util.Date;
88   * import java.util.Locale;
89   * import javax.xml.bind.annotation.XmlElement;
90   * import javax.xml.bind.annotation.XmlRootElement;
91   * &#64;XmlRootElement
92   * public final class Page {
93   *   &#64;XmlElement
94   *   public String version() {
95   *     return Manifests.read("Foo-Version");
96   *   }
97   *   &#64;XmlElement
98   *   public Date date() {
99   *     return new SimpleDateFormat("yyyy.MM.dd", Locale.ENGLISH).parse(
100  *       Manifests.read("Foo-Date");
101  *     );
102  *   }
103  * }</pre>
104  *
105  * <p>In unit and integration tests you may need to inject some values
106  * to {@code MANIFEST.MF} in runtime (for example, in your bootstrap Groovy
107  * scripts):
108  *
109  * <pre> import com.jcabi.manifests.Manifests
110  * Manifests.inject("Foo-URL", "http://localhost/abc");</pre>
111  *
112  * <p>When it is necessary to isolate such injections between different unit
113  * tests "snapshots" may help, for example (it's a method in a unit test):
114  *
115  * <pre> &#64;Test
116  * public void testSomeCode() {
117  *   // save current state of all MANIFEST.MF attributes
118  *   final byte[] snapshot = Manifests.snapshot();
119  *   // inject new attribute required for this specific test
120  *   Manifests.inject("Foo-URL", "http://localhost/abc");
121  *   // restore back all attributes, as they were before the injection
122  *   Manifests.revert(snapshot);
123  * }</pre>
124  *
125  * <p>The only dependency you need (check the latest version at
126  * <a href="http://www.jcabi.com/jcabi-manifests/">jcabi-manifests</a>):
127  *
128  * <pre> &lt;dependency>
129  *  &lt;groupId>com.jcabi&lt;/groupId>
130  *  &lt;artifactId>jcabi-manifests&lt;/artifactId>
131  * &lt;/dependency></pre>
132  *
133  * @author Yegor Bugayenko (yegor@tpc2.com)
134  * @version $Id$
135  * @since 0.7
136  * @see <a href="http://download.oracle.com/javase/1,5.0/docs/guide/jar/jar.html#JAR%20Manifest">JAR Manifest</a>
137  * @see <a href="http://maven.apache.org/shared/maven-archiver/index.html">Maven Archiver</a>
138  * @see <a href="http://www.jcabi.com/jcabi-manifests/index.html">www.jcabi.com/jcabi-manifests</a>
139  */

170     /**
171      * Read one attribute available in one of {@code MANIFEST.MF} files.
172      *
173      * <p>If such a attribute doesn't exist {@link IllegalArgumentException}
174      * will be thrown. If you're not sure whether the attribute is present or
175      * not use {@link #exists(String)} beforehand.
176      *
177      * <p>During testing you can inject attributes into this class by means
178      * of {@link #inject(String,String)}.
179      *
180      * <p>The method is thread-safe.
181      *
182      * @param name Name of the attribute
183      * @return The value of the attribute retrieved
184      */
185     public static String read(
186         @NotNull(message = "attribute name can't be NULL")
187         @Pattern(regexp = ".+", message = "attribute name can't be empty")
188         final String name) {
189         if (Manifests.attributes == null) {
190             throw new IllegalArgumentException(
191                 "Manifests haven't been loaded yet, internal error"
192             );
193         }
194         if (!Manifests.exists(name)) {
195             final StringBuilder bldr = new StringBuilder(
196                 Logger.format(
197                     // @checkstyle LineLength (1 line)
198                     "Atribute '%s' not found in MANIFEST.MF file(s) among %d other attribute(s) %[list]s and %d injection(s)",
199                     name,
200                     Manifests.attributes.size(),
201                     new TreeSet<String>(Manifests.attributes.keySet()),
202                     Manifests.INJECTED.size()
203                 )
204             );
205             if (!Manifests.failures.isEmpty()) {
206                 bldr.append("; failures: ").append(
207                     Logger.format("%[list]s", Manifests.failures.keySet())
208                 );
209             }
210             throw new IllegalArgumentException(bldr.toString());
211         }
212         final String result;
213         if (Manifests.INJECTED.containsKey(name)) {
214             result = Manifests.INJECTED.get(name);
215         } else {
216             result = Manifests.attributes.get(name);
217         }
218         Logger.debug(Manifests.class, "#read('%s'): found '%s'", name, result);
219         return result;
220     }

222     /**
223      * Inject new attribute.
224      *
225      * <p>An attribute can be injected in runtime, mostly for the sake of
226      * unit and integration testing. Once injected an attribute becomes
227      * available with {@link #read(String)}.
228      *
229      * <p>The method is thread-safe.
230      *
231      * @param name Name of the attribute
232      * @param value The value of the attribute being injected
233      */
234     public static void inject(
235         @NotNull(message = "injected name can't be NULL")
236         @Pattern(regexp = ".+", message = "name of attribute can't be empty")
237         final String name,
238         @NotNull(message = "inected value can't be NULL") final String value) {
239         if (Manifests.INJECTED.containsKey(name)) {
240             Logger.info(
241                 Manifests.class,
242                 "#inject(%s, '%s'): replaced previous injection '%s'",
243                 name,
244                 value,
245                 Manifests.INJECTED.get(name)
246             );
247         } else {
248             Logger.info(
249                 Manifests.class,
250                 "#inject(%s, '%s'): injected",
251                 name,
252                 value
253             );
254         }
255         Manifests.INJECTED.put(name, value);
256     }

258     /**
259      * Check whether attribute exists in any of {@code MANIFEST.MF} files.
260      *
261      * <p>Use this method before {@link #read(String)} to check whether an
262      * attribute exists, in order to avoid a runtime exception.
263      *
264      * <p>The method is thread-safe.
265      *
266      * @param name Name of the attribute to check
267      * @return Returns {@code TRUE} if it exists, {@code FALSE} otherwise
268      */
269     public static boolean exists(
270         @NotNull(message = "name of attribute can't be NULL")
271         @Pattern(regexp = ".+", message = "name of attribute can't be empty")
272         final String name) {
273         final boolean exists = Manifests.attributes.containsKey(name)
274             || Manifests.INJECTED.containsKey(name);
275         Logger.debug(Manifests.class, "#exists('%s'): %B", name, exists);
276         return exists;
277     }

279     /**
280      * Make a snapshot of current attributes and their values.
281      *
282      * <p>The method is thread-safe.
283      *
284      * @return The snapshot, to be used later with {@link #revert(byte[])}
285      */
286     public static byte[] snapshot() {
287         byte[] snapshot;
288         synchronized (Manifests.INJECTED) {
289             snapshot = SerializationUtils.serialize(
290                 (Serializable) Manifests.INJECTED
291             );
292         }
293         Logger.debug(
294             Manifests.class,
295             "#snapshot(): created (%d bytes)",
296             snapshot.length
297         );
298         return snapshot;
299     }
300 
301     /**
302      * Revert to the state that was recorded by {@link #snapshot()}.
303      *
304      * <p>The method is thread-safe.
305      *
306      * @param snapshot The snapshot taken by {@link #snapshot()}
307      */
308     @SuppressWarnings("unchecked")
309     public static void revert(@NotNull(message = "snapshot can't be NULL")
310         final byte[] snapshot) {
311         synchronized (Manifests.INJECTED) {
312             Manifests.INJECTED.clear();
313             Manifests.INJECTED.putAll(
314                 (Map<String, String>) SerializationUtils.deserialize(snapshot)
315             );
316         }
317         Logger.debug(
318             Manifests.class,
319             "#revert(%d bytes): reverted",
320             snapshot.length
321         );
322     }

324     /**
325      * Append attributes from the web application {@code MANIFEST.MF}.
326      *
327      * <p>You can call this method in your own
328      * {@link javax.servlet.Filter} or
329      * {@link javax.servlet.ServletContextListener},
330      * in order to inject {@code MANIFEST.MF} attributes to the class.
331      *
332      * <p>The method is thread-safe.
333      *
334      * @param ctx Servlet context
335      * @see #Manifests()
336      * @throws IOException If some I/O problem inside
337      */
338     public static void append(
339         @NotNull(message = "servlet context can't be NULL")
340         final ServletContext ctx)
341         throws IOException {
342         final URL main;
343         try {
344             main = ctx.getResource("/META-INF/MANIFEST.MF");
345         } catch (final MalformedURLException ex) {
346             throw new IOException(ex);
347         }
348         if (main == null) {
349             Logger.warn(
350                 Manifests.class,
351                 "#append(%s): MANIFEST.MF not found in WAR package",
352                 ctx.getClass().getName()
353             );
354         } else {
355             final long start = System.currentTimeMillis();
356             final Map<String, String> attrs = Manifests.load(main);
357             Manifests.attributes.putAll(attrs);
358             Logger.info(
359                 Manifests.class,
360                 // @checkstyle LineLength (1 line)
361                 "#append(%s): %d attribs loaded from %s in %[ms]s (%d total): %[list]s",
362                 ctx.getClass().getName(),
363                 attrs.size(),
364                 main,
365                 System.currentTimeMillis() - start,
366                 Manifests.attributes.size(),
367                 new TreeSet<String>(attrs.keySet())
368             );
369         }
370     }

372     /**
373      * Append attributes from the file.
374      *
375      * <p>The method is thread-safe.
376      *
377      * @param file The file to load attributes from
378      * @throws IOException If some I/O problem inside
379      */
380     @SuppressWarnings("PMD.PrematureDeclaration")
381     public static void append(
382         @NotNull(message = "file can't be NULL")
383         final File file) throws IOException {
384         final Map<String, String> attrs;
385         final long start = System.currentTimeMillis();
386         try {
387             attrs = Manifests.load(file.toURI().toURL());
388         } catch (final MalformedURLException ex) {
389             throw new IOException(ex);
390         }
391         Manifests.attributes.putAll(attrs);
392         Logger.info(
393             Manifests.class,
394             // @checkstyle LineLength (1 line)
395             "#append('%s'): %d attributes loaded in %[ms]s (%d total): %[list]s",
396             file, attrs.size(),
397             System.currentTimeMillis() - start,
398             Manifests.attributes.size(),
399             new TreeSet<String>(attrs.keySet())
400         );
401     }

428     /**
429      * Load attributes from classpath.
430      *
431      * <p>This method doesn't throw any checked exceptions because it is called
432      * from a static context above. It's just more convenient to catch all
433      * exceptions here than above in a static call block.
434      *
435      * @return All found attributes
436      */
437     private static Map<String, String> load() {
438         final long start = System.currentTimeMillis();
439         Manifests.failures = new ConcurrentHashMap<URI, String>(0);
440         final Map<String, String> attrs =
441             new ConcurrentHashMap<String, String>(0);
442         int count = 0;
443         for (final URI uri : Manifests.uris()) {
444             try {
445                 attrs.putAll(Manifests.load(uri.toURL()));
446             } catch (final IOException ex) {
447                 Manifests.failures.put(uri, ex.getMessage());
448                 Logger.error(
449                     Manifests.class,
450                     "#load(): '%s' failed %[exception]s",
451                     uri, ex
452                 );
453             }
454             ++count;
455         }
456         Logger.info(
457             Manifests.class,
458             "#load(): %d attribs loaded from %d URL(s) in %[ms]s: %[list]s",
459             attrs.size(), count,
460             System.currentTimeMillis() - start,
461             new TreeSet<String>(attrs.keySet())
462         );
463         return attrs;
464     }

466     /**
467      * Find all URLs.
468      *
469      * <p>This method doesn't throw any checked exceptions just for convenience
470      * of calling of it (above in {@link #load}), although it is clear that
471      * {@link IOException} is a good candidate for being thrown out of it.
472      *
473      * @return The list of URLs
474      * @see #load()
475      */
476     private static Set<URI> uris() {
477         final Enumeration<URL> resources;
478         try {
479             resources = Thread.currentThread().getContextClassLoader()
480                 .getResources("META-INF/MANIFEST.MF");
481         } catch (final IOException ex) {
482             throw new IllegalStateException(ex);
483         }
484         final Set<URI> uris = new HashSet<URI>(0);
485         while (resources.hasMoreElements()) {
486             try {
487                 uris.add(resources.nextElement().toURI());
488             } catch (final URISyntaxException ex) {
489                 throw new IllegalStateException(ex);
490             }
491         }
492         return uris;
493     }

496      * Load attributes from one file.
497      * @param url The URL of it
498      * @return The attributes loaded
499      * @see #load()
500      * @throws IOException If some problem happens
501      */
502     private static Map<String, String> load(final URL url)
503         throws IOException {
504         return Manifests.load(url.openStream());
505     }

507     /**
508      * Load attributes from input stream.
509      *
510      * <p>Inside the method we catch {@code RuntimeException} (which may look
511      * suspicious) in order to protect our execution flow from expected (!)
512      * exceptions from {@link Manifest#getMainAttributes()}. For some reason,
513      * this JDK method doesn't throw checked exceptions if {@code MANIFEST.MF}
514      * file format is broken. Instead, it throws a runtime exception (an
515      * unchecked one), which we should catch in such an inconvenient way.
516      *
517      * @param stream Stream to load from
518      * @return The attributes loaded
519      * @see #load()
520      * @throws IOException If some problem happens
521      * @since 0.8
522      */
523     @SuppressWarnings("PMD.AvoidCatchingGenericException")
524     private static Map<String, String> load(final InputStream stream)
525         throws IOException {
526         final Map<String, String> props =
527             new ConcurrentHashMap<String, String>(0);
528         try {
529             final Manifest manifest = new Manifest(stream);
530             final Attributes attrs = manifest.getMainAttributes();
531             for (final Object key : attrs.keySet()) {
532                 final String value = attrs.getValue(
533                     Attributes.Name.class.cast(key)
534                 );
535                 props.put(key.toString(), value);
536             }
537             Logger.debug(
538                 Manifests.class,
539                 "#load(): %d attributes loaded (%[list]s)",
540                 props.size(), new TreeSet<String>(props.keySet())
541             );
542         // @checkstyle IllegalCatch (1 line)
543         } catch (final RuntimeException ex) {
544             Logger.error(Manifests.class, "#load(): failed %[exception]s", ex);
545         } finally {
546             IOUtils.closeQuietly(stream);
547         }
548         return props;
549     }
